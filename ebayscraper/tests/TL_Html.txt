			function doWatchAuction() {

				var wa = document.getElementById("alNextWatchAction");

				var currOptionCode = document.getElementById("auctionOptionCode").value;

				var getNewOptionCode = 0;

				var newOptionCode = new Array();

	

				$.ajax({

					async: false,

					type: 'POST',

					url: '/WatchAuctionProcess.cfm',

					dataType: 'json',

					data: {

						waItemID: 1799983,

						waAction: wa.value,

						procOptionCode: currOptionCode

					},

					cache: false,

					success: function (data, status) {

						if (data.isSuccess == 1) {

							//Sucess

							document.getElementById("btnWatchAuction").src = '/images/' + data.watchImage;

							document.getElementById("valUsersWatching").innerHTML = data.watchCnt;

							wa.value = data.nextAction;

							getNewOptionCode = 1;

						} else {

							//Failure

							if (data.error != '') {

								alert(data.error);

							}

							//reload to generate new token value and clear any problems

							location.reload();

						}

					},

					error: function (data, status) {

						var tempError = "An error occurred processing your watch for this auction.";

						alert(tempError);

						//reload to generate new token value and clear any problems

						location.reload();

					}

				});



				if (getNewOptionCode == 1) {

					newOptionCode = changeOptionCode(1);

					if (newOptionCode[0] == 0 || newOptionCode[1] == '') {

						//error so reload to generate new token value and clear any problems

						location.reload();

					}

					else {

						document.getElementById("auctionOptionCode").value = newOptionCode[1];

					}

				}

			}

			

		

		

		function validateBidConfirmForm() {

			var bcMaxBid = document.getElementById("bcMaxBid");

			var valbcMaxBid = bcMaxBid.value * 1;

			var valbcSealedBid = document.getElementById("hidbcSealedBid").value;

			var valbcShowDeliveryMethod = document.getElementById("hidbcShowDeliveryMethod").value;

			var valbcDeliveryOption = document.getElementById("hidbcDeliveryOption").value;

			var valbcMinToBid = document.getElementById("hidbcMinToBid").value * 1;

			var valbcMinToBidText = document.getElementById("hidbcMinToBidText").value;

			var valMinSealedBid = 102 * 1;

				

			valbcMaxBid = removeComma(bcMaxBid.value);

	

			if (valbcMaxBid == '') {

				alert("Please enter a Max. Bid.");

				bcMaxBid.focus();

				return false;

			}

	

			if (isNaN(valbcMaxBid) || valbcMaxBid == 0) {

				alert("Please enter a valid Max. Bid.");

				bcMaxBid.focus();

				return false;

			}

	

			if (valbcSealedBid == 1) {

				if (valbcMaxBid < valMinSealedBid) {

					alert("Your bid is below the minimum bid for this auction.  Please enter a higher amount.");

					bcMaxBid.focus();

					return false;

				}

			} else {

				if (valbcMaxBid < valbcMinToBid) {

					valbcMinToBid = valbcMinToBid * 1;

					valbcMinToBid = valbcMinToBid.toFixed(2);

					alert("Please enter a Max. Bid of at least $" + valbcMinToBidText);

					bcMaxBid.focus();

					return false;

				}

			}



			if (!isRadioSelected("bcBBYDelivery",false)) {

				alert("Please select a delivery method.");

				return false;

			}



			if (valbcShowDeliveryMethod != 0) {

				if (isRadioSelected("bcBBYDelivery",true) == 1) {

					if (!isRadioSelected("bcLiftGate",false) || !isRadioSelected("bcInsideDelivery",false)) {

						var msg = "You must answer all delivery service questions.";

						if (document.getElementById('LiftGate2').style.display == "none") msg = "You must answer the Inside Delivery service question.";

						alert(msg);

						return false;

					}

				}

			}

	

			return true;

		}
